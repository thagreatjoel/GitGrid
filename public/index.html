/api/activity

Get recent GitHub activity of a user.

GET /api/activity?user=thagreatjoel

{
  "user": "thagreatjoel",
  "recentActivity": [
    {
      "type": "PushEvent",
      "repo": "joel/gh-graphify",
      "created_at": "2025-07-23T12:45:00Z"
    }
  ]
}

/api/followers

Get follower count of a user.

GET /api/followers?user=thagreatjoel

{ "user": "thagreatjoel", "followers": 42 }

/api/repo

Get all public repositories of a user.

GET /api/repo?user=thagreatjoel

{ "repos": ["gh-graphify", "gh-utils"] }

/api/stats

Get basic stats about a user.

GET /api/stats?user=thagreatjoel

{ "public_repos": 8, "followers": 42 }

/api/lang

Get languages used across repos.

GET /api/lang?user=thagreatjoel

{ "JavaScript": 70, "HTML": 25 }

/api/pin

Get pinned repos of a user.

GET /api/pin?user=thagreatjoel

{ "pinned": ["gh-graphify", "cli-stats"] }

/api/sizes

Get repo sizes in KB.

GET /api/sizes?user=thagreatjoel

{ "gh-graphify": 324, "gh-utils": 192 }

/api/licenses

Get license types used across repos.

GET /api/licenses?user=thagreatjoel

{ "MIT": 4, "GPL": 2 }

/api/gists

Get public gists of a user.

GET /api/gists?user=thagreatjoel

{ "gists": ["https://gist.github.com/..."] }

/api/commits

Get recent commits in a repo.

GET /api/commits?user=thagreatjoel&repo=gh-graphify

{ "repo": "gh-graphify", "commits": [ ... ] }

/api/topics

Get topics/tags of a repo.

GET /api/topics?user=thagreatjoel&repo=gh-graphify

{ "topics": ["api", "github", "nodejs"] }

/api/metrics

Get GitHub user metrics.

GET /api/metrics?user=thagreatjoel

{ "stars": 25, "followers": 42 }

/api/blog

Get blog URL of a GitHub user.

GET /api/blog?user=thagreatjoel

{ "blog": "https://joelblog.vercel.app" }

/api/forkratio

Get ratio of forks to stars.

GET /api/forkratio?user=thagreatjoel

{ "forkRatio": 0.72 }
